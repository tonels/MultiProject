source /etc/profile;

#实际项目路径
appdir=/opt/oes/oes-workbench

#服务名
appname=manage

if [ -d $appdir/$appname/bin ]; then
   cd  $appdir/$appname/bin
    ./service.sh stop
    rm -rf  $appdir/$appname
fi

cd $appdir
cp -f /usr/local/deploy/oes-content-manage-1.0.0.tar.gz oes-content-manage-1.0.0.tar.gz
tar zxvf oes-content-manage-1.0.0.tar.gz
if [[ $? -ne 0 ]];then exit 1;fi

echo '拷贝和解压完成，开始启动服务。。。'

cd  $appdir/$appname/bin
./service.sh start dev
